    94  func (r *IbeamParameterRegistry) getInstanceValues(dpID ibeam_core.DeviceParameterID) (values []*ibeam_core.ParameterValue) {
    95  	deviceIndex := int(dpID.Device) - 1
    96  	parameterIndex := int(dpID.Parameter) - 1
    97  
    98  	r.muValue.RLock()
    99  	if dpID.Device == 0 || dpID.Parameter == 0 || len(r.parameterValue) <= deviceIndex || len(r.parameterValue[deviceIndex]) <= parameterIndex {
   100  		log.Error("Could not get instance values for DeviceParameterID: Device:", dpID.Device, " and param: ", dpID.Parameter)
   101  		r.muValue.RUnlock()
   102  		return nil
   103  	}
   104  	for _, value := range r.parameterValue[deviceIndex][parameterIndex] {
   105  		values = append(values, value.getParameterValue())
   106  	}
   107  	r.muValue.RUnlock()
   108  	return
   109  }

   175  // Get returns the Parameters with their current state for given DeviceParameterIDs.
   176  // If no IDs are given, all Parameters will be returned.
   177  func (s *IbeamServer) Get(_ context.Context, dpIDs *ibeam_core.DeviceParameterIDs) (rParameters *ibeam_core.Parameters, err error) {
   178  	rParameters = &ibeam_core.Parameters{}
   179  	if len(dpIDs.Ids) == 0 {
   180  		for did, dState := range s.parameterRegistry.parameterValue {
   181  			for pid := range dState {
   182  				dpID := ibeam_core.DeviceParameterID{
   183  					Parameter: uint32(pid) + 1,
   184  					Device:    uint32(did) + 1,
   185  				}
   186  				iv := s.parameterRegistry.getInstanceValues(dpID)
   187  				if iv != nil {
   188  					rParameters.Parameters = append(rParameters.Parameters, &ibeam_core.Parameter{
   189  						Id:    &dpID,
   190  						Error: 0,
   191  						Value: iv,
   192  					})
   193  				}
   194  			}
   195  		}
   196  		rParameters.Parameters = append(rParameters.Parameters)
   197  	} else {
   198  		for _, dpID := range dpIDs.Ids {
   199  			if dpID.Device == 0 || dpID.Parameter == 0 {
   200  				err = errors.New("Failed to get instance values " + dpID.String())
   201  				return
   202  			}
   203  			iv := s.parameterRegistry.getInstanceValues(*dpID)
   204  			if err != nil || len(iv) == 0 {
   205  				rParameters.Parameters = append(rParameters.Parameters, &ibeam_core.Parameter{
   206  					Id:    dpID,
   207  					Error: ibeam_core.ParameterError_UnknownError,
   208  					Value: nil,
   209  				})
   210  			} else {
   211  				rParameters.Parameters = append(rParameters.Parameters, &ibeam_core.Parameter{
   212  					Id:    dpID,
   213  					Error: ibeam_core.ParameterError_NoError,
   214  					Value: iv,
   215  				})
   216  			}
   217  		}
   218  	}
   219  	return
   220  }

   268  // Subscribe starts a ServerStream and send updates if a Parameter changes.
   269  // On subscribe all current values should be sent back.
   270  // If no IDs are given, subscribe to every Parameter.
   271  func (s *IbeamServer) Subscribe(dpIDs *ibeam_core.DeviceParameterIDs, stream ibeam_core.IbeamCore_SubscribeServer) error {
   272  	log.Info("New Client subscribed")
   273  	// Fist send all parameters
   274  	parameters, err := s.Get(nil, dpIDs)
   275  	if err != nil {
   276  		return err
   277  	}
   278  	for _, parameter := range parameters.Parameters {
   279  		log.Debugf("Send Parameter with ID '%v' to client", parameter.Id)
   280  		stream.Send(parameter)
   281  	}
   282  
   283  	distributor := make(chan ibeam_core.Parameter, 100)
   284  	s.serverClientsDistributor[distributor] = true
   285  
   286  	log.Debugf("Added distributor number %v", len(s.serverClientsDistributor))
   287  
   288  	ping := time.NewTicker(time.Second)
   289  	for {
   290  		select {
   291  		// TODO check if Stream is closed
   292  		case <-ping.C:
   293  			err := stream.Context().Err()
   294  			if err != nil {
   295  				s.serverClientsDistributor[distributor] = false
   296  				log.Warn("Connection to client for subscription lost")
   297  				return nil
   298  			}
   299  		case parameter := <-distributor:
   300  			if parameter.Id == nil || parameter.Id.Device == 0 || parameter.Id.Parameter == 0 {
   301  				continue
   302  			}
   303  			// Check if Device is Subscribed
   304  			if len(dpIDs.Ids) != 0 && !containsDeviceParameter(parameter.Id, dpIDs) {
   305  				continue
   306  			}
   307  			log.Debugf("Send Parameter with ID '%v' to client from ServerClientsStream", parameter.Id)
   308  			stream.Send(&parameter)
   309  		}
   310  	}
   311  	return nil
   312  }

   322  // CreateServer sets up the ibeam server, parameter manager and parameter registry
   323  func CreateServer(coreInfo ibeam_core.CoreInfo, defaultModel ibeam_core.ModelInfo) (manager *IbeamParameterManager, registry *IbeamParameterRegistry, setToGRPC chan ibeam_core.Parameter, getFromGRPC chan ibeam_core.Parameter) {
   324  	clientsSetter := make(chan ibeam_core.Parameter, 100)
   325  	getFromGRPC = make(chan ibeam_core.Parameter, 100)
   326  	setToGRPC = make(chan ibeam_core.Parameter, 100)
   327  
   328  	fistParameter := ibeam_core.Parameter{}
   329  	fistParameter.Id = &ibeam_core.DeviceParameterID{
   330  		Device:    0,
   331  		Parameter: 0,
   332  	}
   333  
   334  	watcher := make(chan ibeam_core.Parameter)
   335  
   336  	coreInfo.IbeamVersion = ibeam_core.File_ibeam_core_proto.Options().ProtoReflect().Get(ibeam_core.E_IbeamVersion.TypeDescriptor()).String()
   337  
   338  	registry = &IbeamParameterRegistry{
   339  		coreInfo:        coreInfo,
   340  		DeviceInfos:     []*ibeam_core.DeviceInfo{},
   341  		ModelInfos:      []*ibeam_core.ModelInfo{},
   342  		ParameterDetail: [][]*ibeam_core.ParameterDetail{},
   343  		parameterValue:  [][][]*IBeamParameterValueBuffer{},
   344  	}
   345  
   346  	server := IbeamServer{
   347  		parameterRegistry:        registry,
   348  		clientsSetterStream:      clientsSetter,
   349  		serverClientsStream:      watcher,
   350  		serverClientsDistributor: make(map[chan ibeam_core.Parameter]bool),
   351  	}
   352  
   353  	manager = &IbeamParameterManager{
   354  		parameterRegistry:   registry,
   355  		out:                 getFromGRPC,
   356  		in:                  setToGRPC,
   357  		clientsSetterStream: clientsSetter,
   358  		serverClientsStream: watcher,
   359  		server:              &server,
   360  	}
   361  
   362  	go func() {
   363  		for {
   364  			parameter := <-watcher
   365  			for channel, isOpen := range server.serverClientsDistributor {
   366  				if isOpen {
   367  					channel <- parameter
   368  				} else {
   369  					log.Debugf("Deleted Channel %v", channel)
   370  					delete(server.serverClientsDistributor, channel)
   371  				}
   372  			}
   373  		}
   374  	}()
   375  
   376  	log.Info("Server created")
   377  	registry.RegisterModel(&defaultModel)
   378  	return
   379  }

   381  // RegisterParameter registers a Parameter and his Details in the Registry.
   382  func (r *IbeamParameterRegistry) RegisterParameter(detail *ibeam_core.ParameterDetail) uint32 {
   383  	mid := uint32(1)
   384  	if detail.Id != nil {
   385  		if detail.Id.Model != 0 {
   386  			mid = detail.Id.Model
   387  		}
   388  	}
   389  	r.muDetail.RLock()
   390  	if uint32(len(r.ParameterDetail)) <= (mid - 1) {
   391  		log.Panic("Could not register parameter for nonexistent model ", mid)
   392  		return 0
   393  	}
   394  
   395  	modelconfig := &r.ParameterDetail[mid-1]
   396  	paramIndex := uint32(len(*modelconfig) + 1)
   397  	r.muDetail.RUnlock()
   398  	detail.Id = &ibeam_core.ModelParameterID{
   399  		Parameter: paramIndex,
   400  		Model:     mid,
   401  	}
   402  	r.muDetail.Lock()
   403  	*modelconfig = append(*modelconfig, detail)
   404  	r.muDetail.Unlock()
   405  	log.Debugf("ParameterDetail '%v' registered with ID: %v for Model %v", detail.Name, detail.Id.Parameter, detail.Id.Model)
   406  	return paramIndex
   407  }

   435  // RegisterDevice registers a new Device in the Registry with given ModelID
   436  func (r *IbeamParameterRegistry) RegisterDevice(modelID uint32) uint32 { //DeviceInfo) {
   437  	mid := uint32(1)
   438  	if modelID != 0 {
   439  		mid = modelID
   440  	}
   441  	r.muDetail.RLock()
   442  	if uint32(len(r.ParameterDetail)) <= (mid - 1) {
   443  		log.Panicf("Could not register device for nonexistent model with id: %v", mid)
   444  		return 0
   445  	}
   446  
   447  	modelConfig := r.ParameterDetail[mid-1]
   448  	r.muDetail.RUnlock()
   449  
   450  	// create device info
   451  	// take all params from model and generate a value buffer array for all instances
   452  	// add value buffers to the state array
   453  
   454  	parameterValuesBuffer := &[][]*IBeamParameterValueBuffer{}
   455  	for _, parameterDetail := range modelConfig {
   456  		valueInstances := []*IBeamParameterValueBuffer{}
   457  
   458  		// Integer is default
   459  		initialValue := ibeam_core.ParameterValue{Value: &ibeam_core.ParameterValue_Integer{Integer: 0}}
   460  
   461  		switch parameterDetail.ValueType {
   462  		case ibeam_core.ValueType_NoValue:
   463  			initialValue.Value = &ibeam_core.ParameterValue_Cmd{Cmd: ibeam_core.Command_Trigger}
   464  		case ibeam_core.ValueType_Floating:
   465  			initialValue.Value = &ibeam_core.ParameterValue_Floating{Floating: 0.0}
   466  		case ibeam_core.ValueType_Opt:
   467  			initialValue.Value = &ibeam_core.ParameterValue_CurrentOption{CurrentOption: 0}
   468  		case ibeam_core.ValueType_String:
   469  			initialValue.Value = &ibeam_core.ParameterValue_Str{Str: ""}
   470  		case ibeam_core.ValueType_Binary:
   471  			initialValue.Value = &ibeam_core.ParameterValue_Binary{Binary: false}
   472  		}
   473  
   474  		for i := uint32(0); i < parameterDetail.Instances; i++ {
   475  			valueInstances = append(valueInstances, &IBeamParameterValueBuffer{
   476  				instanceID:     i + 1,
   477  				available:      true,
   478  				isAssumedState: true,
   479  				lastUpdate:     time.Now(),
   480  				currentValue:   initialValue,
   481  				targetValue:    initialValue,
   482  			})
   483  		}
   484  
   485  		for i := uint32(0); i < r.coreInfo.MaxDevices; i++ {
   486  			*parameterValuesBuffer = append(*parameterValuesBuffer, valueInstances)
   487  		}
   488  	}
   489  
   490  	r.muInfo.RLock()
   491  	deviceIndex := uint32(len(r.DeviceInfos) + 1)
   492  	r.muInfo.RUnlock()
   493  	r.muInfo.Lock()
   494  	r.DeviceInfos = append(r.DeviceInfos, &ibeam_core.DeviceInfo{
   495  		DeviceID: deviceIndex,
   496  		ModelID:  modelID,
   497  	})
   498  	r.muInfo.Unlock()
   499  	r.muValue.Lock()
   500  	r.parameterValue = append(r.parameterValue, *parameterValuesBuffer)
   501  	r.muValue.Unlock()
   502  
   503  	log.Debugf("Device '%v' registered with model: %v (%v)", deviceIndex, mid, r.ModelInfos[mid-1].Name)
   504  	return deviceIndex
   505  }

   518  // StartWithServer Starts the ibeam parameter routine and the GRPC server in one call. This is blocking and should be called at the end of main
   519  func (m *IbeamParameterManager) StartWithServer(endPoint string) {
   520  	// Start parameter management routine
   521  	m.Start() // TODO: we could use a waitgroup or something here ? cross goroutine error handling is kinda missing
   522  
   523  	lis, err := net.Listen("tcp", endPoint) // TODO: make listen: unix also possible!
   524  	if err != nil {
   525  		log.Fatalf("failed to listen: %v", err)
   526  	}

   529  	grpcServer.Serve(lis)
   530  }

   532  // Start the communication between client and server.
   533  func (m *IbeamParameterManager) Start() {
   534  	go func() {
   535  		for {
   536  			select {
   537  			case parameter := <-m.clientsSetterStream:
   538  				// Client set loop, inputs set requests from grpc to manager
   539  				//parameter := <-m.clientsSetterStream
   540  				state := m.parameterRegistry.parameterValue
   541  				deviceIndex := int(parameter.Id.Device) - 1
   542  				parameterIndex := int(parameter.Id.Parameter) - 1
   543  
   544  				for _, value := range parameter.Value {
   545  					if len(state) <= deviceIndex || len(state[deviceIndex]) <= parameterIndex {
   546  						log.Errorf("Client tried to set invalid device index %v for parameter %v", deviceIndex, parameterIndex+1)
   547  						continue
   548  					}
   549  					if value.InstanceID == 0 || len(state[deviceIndex][parameterIndex]) < int(value.InstanceID) {
   550  						log.Errorf("Received invalid instance id %v for parameter %v", value.InstanceID, parameterIndex+1)
   551  						continue
   552  					}
   553  					parameterBuffer := state[deviceIndex][parameterIndex][value.InstanceID-1]
   554  					if value.Value == nil {
   555  						continue
   556  					}
   557  					switch v := value.Value.(type) {
   558  					case *ibeam_core.ParameterValue_Cmd:
   559  						log.Debugf("Got Set CMD: %v", v)
   560  						m.out <- ibeam_core.Parameter{
   561  							Id:    parameter.Id,
   562  							Error: 0,
   563  							Value: []*ibeam_core.ParameterValue{value},
   564  						}
   565  					case *ibeam_core.ParameterValue_Binary:
   566  						log.Debugf("Got Set Binary: %v", v)
   567  						if v != parameterBuffer.currentValue.Value {
   568  							parameterBuffer.isAssumedState = true
   569  						}
   570  						parameterBuffer.targetValue.Value = v
   571  						parameterBuffer.tryCount = 0
   572  					case *ibeam_core.ParameterValue_Floating:
   573  						log.Debugf("Got Set Float: %v", v)
   574  						modelIndex := m.parameterRegistry.getModelIndex(deviceIndex)
   575  						parameterConfig := m.parameterRegistry.ParameterDetail[modelIndex][parameterIndex]
   576  						if v.Floating > parameterConfig.Maximum {
   577  							log.Debugf("Max violation for parameter %v", parameterIndex+1)
   578  							m.serverClientsStream <- ibeam_core.Parameter{
   579  								Id:    parameter.Id,
   580  								Error: ibeam_core.ParameterError_MaxViolation,
   581  								Value: []*ibeam_core.ParameterValue{},
   582  							}
   583  							continue
   584  						}
   585  						if v.Floating < parameterConfig.Minimum {
   586  							log.Debugf("Min violation for parameter %v", parameterIndex+1)
   587  							m.serverClientsStream <- ibeam_core.Parameter{
   588  								Id:    parameter.Id,
   589  								Error: ibeam_core.ParameterError_MinViolation,
   590  								Value: []*ibeam_core.ParameterValue{},
   591  							}
   592  							continue
   593  						}
   594  						parameterBuffer.targetValue.Value = v
   595  						parameterBuffer.tryCount = 0
   596  
   597  					case *ibeam_core.ParameterValue_Integer:
   598  						log.Debugf("Got Set Integer: %v", v)
   599  						modelIndex := m.parameterRegistry.getModelIndex(deviceIndex)
   600  						parameterConfig := m.parameterRegistry.ParameterDetail[modelIndex][parameterIndex]
   601  						if v.Integer > int32(parameterConfig.Maximum) {
   602  							log.Debugf("Max violation for parameter %v", parameterIndex+1)
   603  							m.serverClientsStream <- ibeam_core.Parameter{
   604  								Id:    parameter.Id,
   605  								Error: ibeam_core.ParameterError_MaxViolation,
   606  								Value: []*ibeam_core.ParameterValue{},
   607  							}
   608  							continue
   609  						}
   610  						if v.Integer < int32(parameterConfig.Minimum) {
   611  							log.Debugf("Min violation for parameter %v", parameterIndex+1)
   612  							m.serverClientsStream <- ibeam_core.Parameter{
   613  								Id:    parameter.Id,
   614  								Error: ibeam_core.ParameterError_MinViolation,
   615  								Value: []*ibeam_core.ParameterValue{},
   616  							}
   617  							continue
   618  						}
   619  
   620  						parameterBuffer.targetValue.Value = v
   621  						parameterBuffer.tryCount = 0
   622  
   623  					case *ibeam_core.ParameterValue_OptionList:
   624  						log.Debugf("Got Set Option List: %v", v)
   625  
   626  					case *ibeam_core.ParameterValue_Str:
   627  						log.Debugf("Got Set String: %v", v)
   628  					case *ibeam_core.ParameterValue_CurrentOption:
   629  						log.Debugf("Got Set Current Option: %v", v)
   630  
   631  						modelIndex := m.parameterRegistry.getModelIndex(deviceIndex)
   632  						parameterConfig := m.parameterRegistry.ParameterDetail[modelIndex][parameterIndex]
   633  						if parameterConfig.OptionList == nil {
   634  							log.Errorf("No option List found for Parameter %v", v)
   635  							continue
   636  						}
   637  						if v.CurrentOption > uint32(len(parameterConfig.OptionList.Options)) {
   638  							log.Errorf("Invalid operation index for parameter %v", parameterIndex+1)
   639  							m.serverClientsStream <- ibeam_core.Parameter{
   640  								Id:    parameter.Id,
   641  								Error: ibeam_core.ParameterError_UnknownID,
   642  								Value: []*ibeam_core.ParameterValue{},
   643  							}
   644  							continue
   645  						}
   646  
   647  						parameterBuffer.targetValue.Value = v
   648  						parameterBuffer.tryCount = 0
   649  
   650  					}
   651  				}
   652  
   653  			case parameter := <-m.in:
   654  				// Param Ingest loop, inputs changes from camera to manager
   655  				if parameter.Id == nil {
   656  					continue
   657  				}
   658  				state := m.parameterRegistry.parameterValue
   659  				deviceIndex := int(parameter.Id.Device) - 1
   660  				parameterIndex := int(parameter.Id.Parameter) - 1
   661  				modelIndex := int(m.parameterRegistry.getModelIndex(deviceIndex))
   662  				parameterConfig := *m.parameterRegistry.ParameterDetail[modelIndex][parameterIndex]
   663  				shouldSend := false
   664  				for _, value := range parameter.Value {
   665  					if len(state) <= deviceIndex || len(state[deviceIndex]) <= parameterIndex {
   666  						log.Errorf("Client tried to set invalid device %v for parameter %v", deviceIndex+1, parameterIndex+1)
   667  						continue
   668  					}
   669  					if len(state[deviceIndex][parameterIndex]) <= int(value.InstanceID-1) {
   670  						log.Errorf("Received invalid instance id %v for parameter %v", value.InstanceID, parameterIndex)
   671  						continue
   672  					}
   673  					parameterBuffer := state[deviceIndex][parameterIndex][value.InstanceID-1]
   674  					if value.Value != nil {
   675  						if time.Until(parameterBuffer.lastUpdate).Milliseconds() > int64(parameterConfig.QuarantineDelayMs) {
   676  							parameterBuffer.targetValue.Value = value.Value
   677  						}
   678  						// If Type of Parameter is Opt, find the right Opt
   679  						switch parameterConfig.ValueType {
   680  						case ibeam_core.ValueType_Opt:
   681  							switch v := value.Value.(type) {
   682  							case *ibeam_core.ParameterValue_Str:
   683  								id, err := getIDFromOptionListByElementName(*parameterConfig.OptionList, v.Str)
   684  								if err != nil {
   685  									log.Error(err)
   686  									// TODO get new option list maybe
   687  									continue
   688  								}
   689  								parameterBuffer.targetValue = ibeam_core.ParameterValue{
   690  									Value: &ibeam_core.ParameterValue_CurrentOption{
   691  										CurrentOption: id,
   692  									},
   693  								}
   694  							case *ibeam_core.ParameterValue_CurrentOption:
   695  								parameterBuffer.targetValue.Value = v
   696  							case *ibeam_core.ParameterValue_OptionList:
   697  								if !parameterConfig.OptionListIsDynamic {
   698  									log.Errorf("Parameter with ID %v has no Dynamic OptionList", parameter.Id.Parameter)
   699  									continue
   700  								}
   701  								parameterBuffer.targetValue.Value = v
   702  							default:
   703  								log.Errorf("Valuetype of Parameter is Opt and so we should get a String or Opt, but got %T", value)
   704  								continue
   705  							}
   706  						case ibeam_core.ValueType_Binary:
   707  							if _, ok := value.Value.(*ibeam_core.ParameterValue_Binary); !ok {
   708  								log.Errorf("Parameter with ID %v is Type Binary but got %T", parameter.Id.Parameter, parameterConfig.ValueType)
   709  								continue
   710  							}
   711  						case ibeam_core.ValueType_Floating:
   712  							if _, ok := value.Value.(*ibeam_core.ParameterValue_Floating); !ok {
   713  								log.Errorf("Parameter with ID %v is Type Float but got %T", parameter.Id.Parameter, parameterConfig.ValueType)
   714  								continue
   715  							}
   716  						case ibeam_core.ValueType_Integer:
   717  							if _, ok := value.Value.(*ibeam_core.ParameterValue_Integer); !ok {
   718  								log.Errorf("Parameter with ID %v is Type Integer but got %T", parameter.Id.Parameter, parameterConfig.ValueType)
   719  								continue
   720  							}
   721  						case ibeam_core.ValueType_String:
   722  							if _, ok := value.Value.(*ibeam_core.ParameterValue_Str); !ok {
   723  								log.Errorf("Parameter with ID %v is Type String but got %T", parameter.Id.Parameter, parameterConfig.ValueType)
   724  								continue
   725  							}
   726  						case ibeam_core.ValueType_NoValue:
   727  							log.Errorf("Parameter with ID %v has No Value but got %T", parameter.Id.Parameter, parameterConfig.ValueType)
   728  							continue
   729  						}
   730  
   731  						parameterBuffer.currentValue.Value = value.Value
   732  						parameterBuffer.isAssumedState = parameterBuffer.currentValue.Value != parameterBuffer.targetValue.Value
   733  					} else {
   734  						parameterBuffer.available = value.Available
   735  					}
   736  					shouldSend = true
   737  				}
   738  
   739  				if !shouldSend {
   740  					continue
   741  				}
   742  
   743  				if values := m.parameterRegistry.getInstanceValues(*parameter.GetId()); values != nil {
   744  					m.serverClientsStream <- ibeam_core.Parameter{Value: values, Id: parameter.Id, Error: 0}
   745  				}
   746  
   747  			}
   748  
   749  			// Main Loop
   750  			state := m.parameterRegistry.parameterValue
   751  			for _, device := range m.parameterRegistry.ParameterDetail {
   752  				for _, parameterDetail := range device {
   753  					if ibeam_core.ControlStyle_Undefined == parameterDetail.ControlStyle {
   754  						continue
   755  					}
   756  
   757  					devices := m.parameterRegistry.parameterValue
   758  					for did := 0; did < len(devices); did++ {
   759  						for iid := 0; iid < int(parameterDetail.Instances); iid++ {
   760  							if parameterDetail.Id == nil {
   761  								continue
   762  							}
   763  
   764  							parameterBuffer := state[did][parameterDetail.Id.Parameter-1][iid]
   765  
   766  							switch parameterDetail.ControlStyle {
   767  							case ibeam_core.ControlStyle_Normal:
   768  								if parameterBuffer.currentValue.Value == parameterBuffer.targetValue.Value {
   769  									continue
   770  								}
   771  								if time.Until(parameterBuffer.lastUpdate).Milliseconds()*(-1) < int64(parameterDetail.ControlDelayMs) {
   772  									continue
   773  								}
   774  								if parameterDetail.FeedbackStyle == ibeam_core.FeedbackStyle_NoFeedback {
   775  									continue
   776  								}
   777  								if parameterDetail.RetryCount != 0 {
   778  									parameterBuffer.tryCount++
   779  									if parameterBuffer.tryCount > parameterDetail.RetryCount {
   780  										log.Errorf("Failed to set parameter %v in %v tries on device %v", parameterDetail.Id.Parameter, parameterDetail.RetryCount, did+1)
   781  										parameterBuffer.targetValue = parameterBuffer.currentValue
   782  										parameterBuffer.isAssumedState = false
   783  
   784  										m.serverClientsStream <- ibeam_core.Parameter{
   785  											Value: []*ibeam_core.ParameterValue{parameterBuffer.getParameterValue()},
   786  											Id: &ibeam_core.DeviceParameterID{
   787  												Device:    uint32(did + 1),
   788  												Parameter: parameterDetail.Id.Parameter,
   789  											},
   790  											Error: ibeam_core.ParameterError_MaxRetrys,
   791  										}
   792  										continue
   793  									}
   794  								}
   795  
   796  								parameterBuffer.lastUpdate = time.Now()
   797  								parameterBuffer.isAssumedState = true
   798  								if parameterDetail.Id == nil {
   799  									log.Error("No Parameter provided")
   800  									continue
   801  								}
   802  
   803  								// If we Have a current Option, get the Value for the option from the Option List
   804  								if value, ok := parameterBuffer.targetValue.Value.(*ibeam_core.ParameterValue_CurrentOption); ok {
   805  									name, err := getElementNameFromOptionListByID(*parameterDetail.OptionList, *value)
   806  									if err != nil {
   807  										log.Error(err)
   808  										continue
   809  									}
   810  									parameterBuffer.targetValue.Value = &ibeam_core.ParameterValue_Str{Str: name}
   811  								}
   812  
   813  								m.out <- ibeam_core.Parameter{
   814  									Value: []*ibeam_core.ParameterValue{parameterBuffer.getParameterValue()},
   815  									Id: &ibeam_core.DeviceParameterID{
   816  										Device:    uint32(did + 1),
   817  										Parameter: parameterDetail.Id.Parameter,
   818  									},
   819  									Error: 0,
   820  								}
   821  
   822  							case ibeam_core.ControlStyle_ControlledIncremental:
   823  								if parameterDetail.FeedbackStyle == ibeam_core.FeedbackStyle_NoFeedback ||
   824  									parameterBuffer.currentValue.Value == parameterBuffer.targetValue.Value ||
   825  									time.Until(parameterBuffer.lastUpdate).Milliseconds() < int64(parameterDetail.ControlDelayMs) {
   826  									continue
   827  								}
   828  
   829  								if parameterDetail.RetryCount != 0 {
   830  									parameterBuffer.tryCount++
   831  									if parameterBuffer.tryCount > parameterDetail.RetryCount {
   832  										log.Errorf("Failed to set parameter %v in %v tries on device %v", parameterDetail.Id.Parameter, parameterDetail.RetryCount, did+1)
   833  										parameterBuffer.targetValue = parameterBuffer.currentValue
   834  										continue
   835  									}
   836  								}
   837  								type action string
   838  								const (
   839  									// Increment ...
   840  									Increment action = "Increment"
   841  									// Decrement ...
   842  									Decrement action = "Decrement"
   843  									// NoOperation ...
   844  									NoOperation action = "NoOperation"
   845  								)
   846  
   847  								var cmdAction action
   848  
   849  								switch value := parameterBuffer.currentValue.Value.(type) {
   850  								case *ibeam_core.ParameterValue_Integer:
   851  									if targetValue, ok := parameterBuffer.targetValue.Value.(*ibeam_core.ParameterValue_Integer); ok {
   852  										if value.Integer < targetValue.Integer {
   853  											cmdAction = Increment
   854  										} else if value.Integer > targetValue.Integer {
   855  											cmdAction = Decrement
   856  										}
   857  									}
   858  								case *ibeam_core.ParameterValue_Floating:
   859  									if targetValue, ok := parameterBuffer.targetValue.Value.(*ibeam_core.ParameterValue_Floating); ok {
   860  										if value.Floating < targetValue.Floating {
   861  											cmdAction = Increment
   862  										} else if value.Floating > targetValue.Floating {
   863  											cmdAction = Decrement
   864  										}
   865  									}
   866  								case *ibeam_core.ParameterValue_CurrentOption:
   867  									if targetValue, ok := parameterBuffer.targetValue.Value.(*ibeam_core.ParameterValue_CurrentOption); ok {
   868  										if value.CurrentOption < targetValue.CurrentOption {
   869  											cmdAction = Increment
   870  										} else if value.CurrentOption > targetValue.CurrentOption {
   871  											cmdAction = Decrement
   872  										}
   873  									}
   874  								default:
   875  									log.Errorf("Could not match Valuetype %T", value)
   876  								}
   877  
   878  								var cmdValue []*ibeam_core.ParameterValue
   879  
   880  								switch cmdAction {
   881  								case Increment:
   882  									cmdValue = append(cmdValue, parameterBuffer.incrementParameterValue())
   883  								case Decrement:
   884  									cmdValue = append(cmdValue, parameterBuffer.decrementParameterValue())
   885  								case NoOperation:
   886  									continue
   887  								}
   888  								parameterBuffer.lastUpdate = time.Now()
   889  								if parameterDetail.Id == nil {
   890  									log.Errorf("No DeviceParameterID")
   891  									continue
   892  								}
   893  
   894  								m.out <- ibeam_core.Parameter{
   895  									Id: &ibeam_core.DeviceParameterID{
   896  										Device:    uint32(did + 1),
   897  										Parameter: parameterDetail.Id.Parameter,
   898  									},
   899  									Error: 0,
   900  									Value: cmdValue,
   901  								}
   902  							case ibeam_core.ControlStyle_NoControl, ibeam_core.ControlStyle_Incremental, ibeam_core.ControlStyle_Oneshot:
   903  								// DO Nothing
   904  							default:
   905  								log.Errorf("Could not match controlltype")
   906  								continue
   907  							}
   908  						}
   909  					}
   910  				}
   911  			}
   912  		}
   913  	}()
