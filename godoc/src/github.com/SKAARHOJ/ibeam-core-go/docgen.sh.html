     1#!/bin/sh
     2
     3GO_MODULE=github.com/SKAARHOJ/ibeam-core-go
     4
     5function normalize_url {
     6    # Normalize provided URL. Removing double slashes
     7    echo "$1" | sed -E 's,([^:]/)/+,\1,g'
     8}
     9
    10# Setup
    11rm -rf "${GO_DOC_HTML_OUTPUT:-godoc}"
    12
    13# Extract Go module name from a Go module file
    14if [[ -z "$GO_MODULE" ]]; then
    15    if [[ -z "$1" ]]; then
    16      echo "Please specify a go module"
    17      exit
    18    else
    19        GO_MODULE=$1
    20    fi
    21fi
    22
    23
    24# URL path to Go package and module documentation
    25URL=$(normalize_url "http://${GO_DOC_HTTP:-localhost:6060}/pkg/$GO_MODULE/")
    26
    27# Starting godoc server
    28echo "Starting godoc server..."
    29godoc -http="${GO_DOC_HTTP:-localhost:6060}" &
    30PID=$!
    31
    32# Waiting for godoc server
    33while ! curl --fail --silent "$URL" 2>&1 >/dev/null; do
    34    sleep 0.1
    35done
    36
    37# Download all documentation content from running godoc server
    38wget \
    39    --recursive \
    40    --no-verbose \
    41    --convert-links \
    42    --page-requisites \
    43    --adjust-extension \
    44    --execute=robots=off \
    45    --include-directories="/lib,/pkg/$GO_MODULE,/src/$GO_MODULE" \
    46    --exclude-directories="*" \
    47    --directory-prefix="${GO_DOC_HTML_OUTPUT:-godoc}" \
    48    --no-host-directories \
    49    "$URL"
    50
    51# Generate a simple redirect
    52echo "<script>window.location.href = '/pkg/$GO_MODULE';</script>" > ${GO_DOC_HTML_OUTPUT:-godoc}/index.html
    53
    54# Stop godoc server
    55kill -9 "$PID"
    56echo "Stopped godoc server"
    57echo "Go source code documentation generated under ${GO_DOC_HTML_OUTPUT:-godoc}"
