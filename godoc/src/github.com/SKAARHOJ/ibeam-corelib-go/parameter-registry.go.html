   100  // RegisterParameter registers a parameter and its detail struct in the registry.
   101  func (r *IBeamParameterRegistry) RegisterParameter(detail *pb.ParameterDetail) (paramID uint32) {
   102  	r.modelsDone = true
   103  	if r.parametersDone {
   104  		log.Fatal("Can not unregister a parameter after registering the first device")
   105  	}
   106  
   107  	modelID := uint32(0)
   108  	paramID = uint32(0)
   109  	if detail.Id != nil {
   110  		modelID = detail.Id.Model
   111  		paramID = detail.Id.Parameter
   112  	}
   113  	r.muDetail.RLock()
   114  	if modelID == 0 {
   115  		// append to all models, need to check for ids
   116  
   117  		id := r.ParameterIDByName(detail.Name, modelID)
   118  		if id != 0 {
   119  			log.Fatal("Duplicate parameter name for ", detail.Name)
   120  		}
   121  
   122  		defaultModelConfig := r.ParameterDetail[0]
   123  		if paramID == 0 {
   124  			if !r.allowAutoIDs {
   125  				log.Fatalf("Missing ID on parameter '%s'", detail.Name)
   126  			}
   127  			paramID = uint32(len(defaultModelConfig) + 1)
   128  		}
   129  		r.muDetail.RUnlock()
   130  		detail.Id = &pb.ModelParameterID{
   131  			Parameter: paramID,
   132  			Model:     modelID,
   133  		}
   134  
   135  		validateParameter(detail)
   136  
   137  		r.muDetail.Lock()
   138  		for aMid, modelconfig := range r.ParameterDetail {
   139  			dt := proto.Clone(detail).(*pb.ParameterDetail)
   140  			dt.Id.Model = aMid
   141  			modelconfig[paramID] = dt
   142  		}
   143  		r.muDetail.Unlock()
   144  
   145  	} else {
   146  		pid := r.ParameterIDByName(detail.Name, 0)
   147  		if pid != 0 {
   148  			paramID = pid
   149  		}
   150  
   151  		modelconfig, exists := r.ParameterDetail[modelID]
   152  		if !exists {
   153  			log.Fatalf("Could not register parameter '%s' for model with ID: %d", detail.Name, modelID)
   154  		}
   155  		if paramID == 0 {
   156  			if !r.allowAutoIDs {
   157  				log.Fatalf("Missing ID on parameter '%s'", detail.Name)
   158  			}
   159  			paramID = uint32(len(r.ParameterDetail[0]) + 1)
   160  		}
   161  		r.muDetail.RUnlock()
   162  		detail.Id = &pb.ModelParameterID{
   163  			Parameter: paramID,
   164  			Model:     modelID,
   165  		}
   166  
   167  		validateParameter(detail)
   168  		r.muDetail.Lock()
   169  
   170  		modelconfig[paramID] = detail
   171  
   172  		// if the default model does not have the param it still needs to be added there too!
   173  		if pid == 0 {
   174  			dt := proto.Clone(detail).(*pb.ParameterDetail)
   175  			dt.Id.Model = 0
   176  			r.ParameterDetail[0][paramID] = dt
   177  		}
   178  		r.muDetail.Unlock()
   179  		if pid == 0 {
   180  			log.Debugf("ParameterDetail '%v' with ID: %v was overridden for Model %v", detail.Name, detail.Id.Parameter, detail.Id.Model)
   181  			return
   182  		}
   183  	}
   184  
   185  	log.Debugf("ParameterDetail '%v' registered with ID: %v for Model %v", detail.Name, detail.Id.Parameter, detail.Id.Model)
   186  	return
   187  }

   219  // RegisterModel registers a new Model in the Registry with given ModelInfo
   220  func (r *IBeamParameterRegistry) RegisterModel(model *pb.ModelInfo) uint32 {
   221  	if r.modelsDone {
   222  		log.Fatal("Can not register a new model after registering parameters")
   223  	}
   224  
   225  	if model.Name == "" {
   226  		log.Fatal("please specify a name for all models")
   227  	}
   228  
   229  	if model.Description == "" {
   230  		log.Fatal("please specify a description for all models")
   231  	}
   232  
   233  	r.muInfo.Lock()
   234  	if _, exists := r.ModelInfos[model.Id]; exists {
   235  		// if the id already exists count it up
   236  		if !r.allowAutoIDs {
   237  			log.Fatalf("Refusing to autoassign id for model '%s', please specify an explicit ID", model.Name)
   238  		}
   239  		r.muDetail.RLock()
   240  		model.Id = uint32(len(r.ParameterDetail))
   241  		log.Warnf("Autoassigning id %d for model '%s'", model.Id, model.Name)
   242  		r.muDetail.RUnlock()
   243  	}
   244  
   245  	r.ModelInfos[model.Id] = model
   246  	r.muInfo.Unlock()
   247  
   248  	r.muDetail.Lock()
   249  	r.ParameterDetail[model.Id] = map[uint32]*pb.ParameterDetail{}
   250  	r.muDetail.Unlock()
   251  
   252  	log.Debugf("Model '%v' registered with ID: %v ", model.Name, model.Id)
   253  	return model.Id
   254  }

   302  // ReRegisterDevice registers an existing device again. This allows to reset it's state and change the type of model if necessary
   303  // Make sure to handle the error properly
   304  func (r *IBeamParameterRegistry) ReRegisterDevice(deviceID, modelID uint32) error {
   305  	// Check for device exists
   306  	r.muInfo.RLock()
   307  	if _, exists := r.DeviceInfos[deviceID]; exists {
   308  		r.muInfo.RUnlock()
   309  		return fmt.Errorf("Could not re-register device with existing deviceid: %v", deviceID)
   310  	}
   311  	r.muInfo.RUnlock()
   312  
   313  	r.muDetail.RLock()
   314  	if _, exists := r.ParameterDetail[modelID]; !exists {
   315  		r.muDetail.RUnlock()
   316  		log.Fatalf("Could not register device for nonexistent model with id: %v", modelID)
   317  	}
   318  	r.muDetail.RUnlock()
   319  
   320  	// clear device
   321  	r.muInfo.Lock()
   322  	delete(r.DeviceInfos, deviceID)
   323  	r.muInfo.Unlock()
   324  
   325  	r.muValue.Lock()
   326  	delete(r.ParameterValue, deviceID)
   327  	r.muValue.Unlock()
   328  
   329  	// Call register device with old ID
   330  	_, err := r.RegisterDevice(deviceID, modelID)
   331  	return err
   332  }

   334  // RegisterDevice registers a new Device in the Registry with given DeviceID and ModelID. If the DeviceID is 0 it will be picked automatically
   335  // Make sure to handle the error properly
   336  func (r *IBeamParameterRegistry) RegisterDevice(deviceID, modelID uint32) (uint32, error) {
   337  	r.parametersDone = true
   338  
   339  	r.muDetail.RLock()
   340  	defer r.muDetail.RUnlock()
   341  
   342  	if _, exists := r.ParameterDetail[modelID]; !exists {
   343  		return 0, fmt.Errorf("Could not register device for nonexistent model with id: %v", modelID)
   344  	}
   345  
   346  	r.muInfo.RLock()
   347  	if _, exists := r.DeviceInfos[deviceID]; exists {
   348  		r.muInfo.RUnlock()
   349  		return 0, fmt.Errorf("Could not register device with existing deviceid: %v", deviceID)
   350  	}
   351  	r.muInfo.RUnlock()
   352  
   353  	modelConfig := r.ParameterDetail[modelID]
   354  
   355  	// create device info
   356  	// take all params from model and generate a value buffer array for all instances
   357  	// add value buffers to the state array
   358  
   359  	parameterDimensions := map[uint32]*IBeamParameterDimension{}
   360  	for _, parameterDetail := range modelConfig {
   361  		parameterID := parameterDetail.Id.Parameter
   362  
   363  		// Integer is default
   364  		initialValue := &pb.ParameterValue{
   365  			DimensionID:    []uint32{},
   366  			Available:      true,
   367  			IsAssumedState: false,
   368  			Invalid:        true,
   369  			Value:          &pb.ParameterValue_Integer{Integer: 0},
   370  			MetaValues:     map[string]*pb.ParameterMetaValue{},
   371  		}
   372  
   373  		switch parameterDetail.ValueType {
   374  		case pb.ValueType_NoValue:
   375  			initialValue.Value = &pb.ParameterValue_Cmd{Cmd: pb.Command_Trigger}
   376  		case pb.ValueType_Floating:
   377  			initialValue.Value = &pb.ParameterValue_Floating{Floating: 0.0}
   378  		case pb.ValueType_Opt:
   379  			initialValue.Value = &pb.ParameterValue_CurrentOption{CurrentOption: 0}
   380  		case pb.ValueType_String:
   381  			initialValue.Value = &pb.ParameterValue_Str{Str: ""}
   382  		case pb.ValueType_Binary:
   383  			initialValue.Value = &pb.ParameterValue_Binary{Binary: false}
   384  		}
   385  
   386  		if len(parameterDetail.Dimensions) > 3 {
   387  			log.Fatalf("It is not recommended to use more than 3 dimensions, if needed please contact the maintainer")
   388  		}
   389  
   390  		dimensionConfig := []uint32{}
   391  		initialValueDimension := IBeamParameterDimension{
   392  			value: &ibeamParameterValueBuffer{
   393  				dimensionID:    make([]uint32, 0),
   394  				available:      true,
   395  				isAssumedState: true,
   396  				lastUpdate:     time.Now(),
   397  				currentValue:   proto.Clone(initialValue).(*pb.ParameterValue),
   398  				targetValue:    proto.Clone(initialValue).(*pb.ParameterValue),
   399  			},
   400  		}
   401  
   402  		for _, dimension := range parameterDetail.Dimensions {
   403  			dimensionConfig = append(dimensionConfig, dimension.Count)
   404  		}
   405  
   406  		parameterDimensions[parameterID] = generateDimensions(dimensionConfig, &initialValueDimension)
   407  	}
   408  
   409  	r.muInfo.Lock()
   410  	if deviceID == 0 {
   411  		deviceID = uint32(len(r.DeviceInfos) + 1)
   412  		log.Warnf("Automatically assigning DeviceID %d to device with model %d", deviceID, modelID)
   413  	}
   414  	r.DeviceInfos[deviceID] = &pb.DeviceInfo{
   415  		DeviceID: deviceID,
   416  		ModelID:  modelID,
   417  	}
   418  	r.muInfo.Unlock()
   419  
   420  	r.muValue.Lock()
   421  	r.ParameterValue[deviceID] = parameterDimensions
   422  	r.muValue.Unlock()
   423  
   424  	log.Debugf("Device '%v' registered with model: %v (%v)", deviceID, modelID, r.ModelInfos[modelID].Name)
   425  	return deviceID, nil
   426  }

   466  func validateParameter(detail *pb.ParameterDetail) {
   467  	// Fatals
   468  	if detail.Name == "" {
   469  		log.Fatalf("Could not validate parameter ID %v: No name set", detail.Id)
   470  	}
   471  	if detail.ControlStyle == pb.ControlStyle_NoControl && detail.FeedbackStyle == pb.FeedbackStyle_NoFeedback {
   472  		log.Fatalf("Could not validate parameter '%v': Can not have no control and no feedback", detail.Name)
   473  	}
   474  	if detail.ControlStyle == pb.ControlStyle_ControlledIncremental && detail.ValueType != pb.ValueType_Integer {
   475  		log.Fatalf("Could not validate parameter '%v': Controlled Incremental only supported on integers right now", detail.Name)
   476  	}
   477  	if detail.ControlStyle == pb.ControlStyle_Incremental && detail.IncDecStepsLowerLimit == 0 && detail.IncDecStepsUpperLimit == 0 {
   478  		log.Fatalf("Could not validate parameter '%v': Incremental: please provide lower and upper range for incDecSteps", detail.Name)
   479  	}
   480  	if detail.ControlStyle != pb.ControlStyle_Incremental &&
   481  		detail.ControlStyle != pb.ControlStyle_ControlledIncremental &&
   482  		(detail.IncDecStepsLowerLimit != 0 || detail.IncDecStepsUpperLimit != 0) {
   483  		log.Fatalf("Could not validate parameter '%v': Lower and upper limit are only valid on Incremental Control Mode", detail.Name)
   484  	}
   485  	if detail.Label == "" {
   486  		log.Fatalf("Could not validate parameter '%v': No label set", detail.Name)
   487  	}
   488  	if detail.ControlStyle != pb.ControlStyle_NoControl && detail.FeedbackStyle != pb.FeedbackStyle_NoFeedback && detail.RetryCount == 0 {
   489  		log.Fatalf("Parameter '%v': Any non assumed value (FeedbackStyle_NoFeedback) needs to have RetryCount set", detail.Name)
   490  	}
   491  
   492  	// ValueType Checks
   493  	switch detail.ValueType {
   494  	case pb.ValueType_Floating:
   495  		fallthrough
   496  	case pb.ValueType_Integer:
   497  		if detail.Minimum == 0 && detail.Maximum == 0 {
   498  			log.Fatalf("Could not validate parameter '%v': Integer needs min/max set", detail.Name)
   499  		}
   500  	case pb.ValueType_Binary:
   501  		if detail.ControlStyle == pb.ControlStyle_Incremental {
   502  			log.Fatalf("Could not validate parameter '%v': Binary can not have incremental control", detail.Name)
   503  		}
   504  	case pb.ValueType_NoValue:
   505  		if detail.FeedbackStyle != pb.FeedbackStyle_NoFeedback {
   506  			log.Fatalf("Could not validate parameter '%v': NoValue can not have Feedback", detail.Name)
   507  		}
   508  		if detail.Minimum != 0 || detail.Maximum != 0 {
   509  			log.Fatalf("Could not validate parameter '%v': NoValue can not min/max", detail.Name)
   510  		}
   511  	}
   512  
   513  	// Warnings
   514  	if detail.ShortLabel == "" {
   515  		log.Warnf("Parameter '%v': No short label set", detail.Name)
   516  	}
   517  	if detail.Description == "" {
   518  		log.Warnf("Parameter '%v': No description set", detail.Name)
   519  	}
   520  }
