    22  // StartWithServer Starts the ibeam parameter routine and the GRPC server in one call. This is blocking and should be called at the end of main.
    23  // The network must be "tcp", "tcp4", "tcp6", "unix" or "unixpacket".
    24  func (m *IBeamParameterManager) StartWithServer(network, address string) {
    25  	// Start parameter management routine
    26  	m.Start()
    27  
    28  	lis, err := net.Listen(network, address)
    29  	if err != nil {
    30  		log.Fatalf("failed to listen: %v", err)
    31  	}
    32  	grpcServer := grpc.NewServer()
    33  	pb.RegisterIbeamCoreServer(grpcServer, m.server)
    34  	err = grpcServer.Serve(lis)
    35  	if err != nil {
    36  		log.Fatalf("failed to serve: %v", err)
    37  	}
    38  }

    40  func (m *IBeamParameterManager) checkValidParameter(parameter *pb.Parameter) *pb.Parameter {
    41  	// Check if given Parameter has an DeviceParameterID
    42  	if parameter.Id == nil {
    43  		// Client sees what he has send
    44  		parameter.Error = pb.ParameterError_UnknownID
    45  		log.Errorf("Given Parameter %v has no ID", parameter)
    46  
    47  		return &pb.Parameter{
    48  			Id:    parameter.Id,
    49  			Error: pb.ParameterError_UnknownID,
    50  			Value: []*pb.ParameterValue{},
    51  		}
    52  	}
    53  
    54  	// Get Index and ID for Device and Parameter and the actual state of all parameters
    55  	parameterID := parameter.Id.Parameter
    56  	parameterIndex := parameterID
    57  	deviceID := parameter.Id.Device
    58  	modelIndex := m.parameterRegistry.getModelIndex(deviceID)
    59  
    60  	// Get State and the Configuration (Details) of the Parameter, assume mutex is locked in outer layers of parameterLoop
    61  	state := m.parameterRegistry.ParameterValue
    62  	parameterConfig := m.parameterRegistry.ParameterDetail[modelIndex][parameterIndex]
    63  	// Check if device and param id are valid and in the State
    64  	if _, exists := state[deviceID][parameterIndex]; !exists {
    65  		log.Errorf("Invalid ID for: DeviceID %d, ParameterID %d", deviceID, parameterID)
    66  
    67  		return &pb.Parameter{
    68  			Id:    parameter.Id,
    69  			Error: pb.ParameterError_UnknownID,
    70  			Value: []*pb.ParameterValue{},
    71  		}
    72  	}
    73  
    74  	// Check if the configured type of the Parameter has a value
    75  	if parameterConfig.ValueType == pb.ValueType_NoValue && parameterConfig.ControlStyle == pb.ControlStyle_NoControl {
    76  		log.Errorf("Want to set Parameter with ID %v (%v), but it is configured as Type NoValue with no Control", parameterID, parameterConfig.Name)
    77  		return &pb.Parameter{
    78  			Id:    parameter.Id,
    79  			Error: pb.ParameterError_HasNoValue,
    80  			Value: []*pb.ParameterValue{},
    81  		}
    82  	}
    83  
    84  	if parameterConfig.ValueType == pb.ValueType_NoValue && parameterConfig.ControlStyle == pb.ControlStyle_Oneshot {
    85  		if cmd, ok := parameter.Value[0].Value.(*pb.ParameterValue_Cmd); ok {
    86  			if cmd.Cmd != pb.Command_Trigger {
    87  				log.Errorf("Want to set Parameter with ID %v (%v), but it is configured as Type NoValue with ControlStyle OneShot. Accept only Command:Trigger", parameterID, parameterConfig.Name)
    88  				return &pb.Parameter{
    89  					Id:    parameter.Id,
    90  					Error: pb.ParameterError_InvalidType,
    91  					Value: []*pb.ParameterValue{},
    92  				}
    93  			}
    94  		} else {
    95  			log.Errorf("Want to set Parameter with ID %v (%v), but it is configured as Type NoValue with ControlStyle OneShot. Accept only Command:Trigger", parameterID, parameterConfig.Name)
    96  			return &pb.Parameter{
    97  				Id:    parameter.Id,
    98  				Error: pb.ParameterError_InvalidType,
    99  				Value: []*pb.ParameterValue{},
   100  			}
   101  		}
   102  	}
   103  
   104  	return nil
   105  }

   107  // Start the communication between client and server.
   108  func (m *IBeamParameterManager) Start() {
   109  	go func() {
   110  		for {
   111  			// ***************
   112  			//  ClientToManagerLoop, inputs change request from GRPC SET to to manager
   113  			// ***************
   114  		clientToManagerLoop:
   115  			for {
   116  				var parameter *pb.Parameter
   117  				select {
   118  				case parameter = <-m.clientsSetterStream:
   119  					m.ingestTargetParameter(parameter)
   120  				default:
   121  					break clientToManagerLoop
   122  				}
   123  			}
   124  
   125  			// ***************
   126  			//    Param Ingest loop, inputs changes from a device (f.e. a camera) to manager
   127  			// ***************
   128  
   129  		deviceToManagerLoop:
   130  			for {
   131  				var parameter *pb.Parameter
   132  				select {
   133  				case parameter = <-m.in:
   134  					m.ingestCurrentParameter(parameter)
   135  				default:
   136  					break deviceToManagerLoop
   137  				}
   138  			}
   139  
   140  			// ***************
   141  			//    Main Parameter Loop, evaluates all parameter value buffers and sends out necessary changes
   142  			// ***************
   143  
   144  			m.parameterLoop()
   145  			time.Sleep(time.Microsecond * 800)
   146  		}
   147  	}()
   148  }
