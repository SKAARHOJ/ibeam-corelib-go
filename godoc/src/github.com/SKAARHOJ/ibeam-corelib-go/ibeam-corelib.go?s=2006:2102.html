    26  func (s *IbeamServer) GetCoreInfo(_ context.Context, _ *pb.Empty) (*pb.CoreInfo, error) {
    27  	coreInfo := proto.Clone(s.parameterRegistry.coreInfo).(*pb.CoreInfo)
    28  	s.muDistributor.RLock()

    35  // If no IDs are given, all DeviceInfos will be returned.
    36  func (s *IbeamServer) GetDeviceInfo(_ context.Context, deviceIDs *pb.DeviceIDs) (*pb.DeviceInfos, error) {
    37  
    38  	log.Debugf("Client asks for DeviceInfo with ids %v", deviceIDs.Ids)
    39  	s.parameterRegistry.muInfo.RLock()
    40  	defer s.parameterRegistry.muInfo.RUnlock()
    41  
    42  	if len(deviceIDs.Ids) == 0 {
    43  		infos := []*pb.DeviceInfo{}
    44  		for _, info := range s.parameterRegistry.DeviceInfos {
    45  			infos = append(infos, info)
    46  		}
    47  		return &pb.DeviceInfos{DeviceInfos: infos}, nil
    48  	}
    49  
    50  	var rDeviceInfos pb.DeviceInfos
    51  	for _, deviceID := range deviceIDs.Ids {
    52  		_, dExists := s.parameterRegistry.DeviceInfos[deviceID]
    53  		if dExists && deviceID != 0 {
    54  			rDeviceInfos.DeviceInfos = append(rDeviceInfos.DeviceInfos, s.parameterRegistry.DeviceInfos[deviceID])
    55  		}
    56  		// If we have no Device with such a ID, skip
    57  	}
    58  	return &rDeviceInfos, nil
    59

    94  // Get returns the Parameters with their current state for given DeviceParameterIDs.
    95  // If no IDs are given, all Parameters will be returned.
    96  func (s *IbeamServer) Get(_ context.Context, dpIDs *pb.DeviceParameterIDs) (rParameters *pb.Parameters, err error) {
    97  	rParameters = &pb.Parameters{}
    98  	s.parameterRegistry.muValue.RLock()
    99  	defer s.parameterRegistry.muValue.RUnlock()
   100  
   101  	if len(dpIDs.Ids) == 0 {
   102  		for did, dState := range s.parameterRegistry.ParameterValue {
   103  			for pid := range dState {
   104  				dpID := pb.DeviceParameterID{
   105  					Parameter: uint32(pid),
   106  					Device:    uint32(did),
   107  				}
   108  				iv := s.parameterRegistry.getInstanceValues(&dpID)
   109  				if iv != nil {
   110  					rParameters.Parameters = append(rParameters.Parameters, &pb.Parameter{
   111  						Id:    &dpID,
   112  						Error: 0,
   113  						Value: iv,
   114  					})
   115  				}
   116  			}
   117  		}
   118  		rParameters.Parameters = append(rParameters.Parameters)
   119  	} else if len(dpIDs.Ids) == 1 && dpIDs.Ids[0].Parameter == 0 && dpIDs.Ids[0].Device != 0 {
   120  		did := dpIDs.Ids[0].Device
   121  		if _, exists := s.parameterRegistry.ParameterValue[did]; !exists {
   122  			rParameters.Parameters = append(rParameters.Parameters, &pb.Parameter{
   123  				Id:    dpIDs.Ids[0],
   124  				Error: pb.ParameterError_UnknownID,
   125  				Value: nil,
   126  			})
   127  			return
   128  		}
   129  		_, exists := s.parameterRegistry.ParameterValue[did][0]
   130  		log.Info(exists)
   131  		for pid := range s.parameterRegistry.ParameterValue[did] {
   132  			dpID := pb.DeviceParameterID{
   133  				Parameter: pid,
   134  				Device:    did,
   135  			}
   136  			iv := s.parameterRegistry.getInstanceValues(&dpID)
   137  			if iv != nil {
   138  				rParameters.Parameters = append(rParameters.Parameters, &pb.Parameter{
   139  					Id:    &dpID,
   140  					Error: 0,
   141  					Value: iv,
   142  				})
   143  			}
   144  
   145  		}
   146  		rParameters.Parameters = append(rParameters.Parameters)
   147  	} else {
   148  		for _, dpID := range dpIDs.Ids {
   149  			if dpID.Device == 0 || dpID.Parameter == 0 {
   150  				err = errors.New("Failed to get instance values " + dpID.String())
   151  				return
   152  			}
   153  			iv := s.parameterRegistry.getInstanceValues(dpID)
   154  			if len(iv) == 0 {
   155  				rParameters.Parameters = append(rParameters.Parameters, &pb.Parameter{
   156  					Id:    dpID,
   157  					Error: pb.ParameterError_UnknownError,
   158  					Value: nil,
   159  				})
   160  			} else {
   161  				rParameters.Parameters = append(rParameters.Parameters, &pb.Parameter{
   162  					Id:    dpID,
   163  					Error: pb.ParameterError_NoError,
   164  					Value: iv,
   165  				})
   166  			}
   167  		}
   168  	}
   169  	return
   170  }

   172  // GetParameterDetails returns the Details for given ModelParameterIDs.
   173  // If no IDs are given, the Details from all Parameters will be returned.
   174  func (s *IbeamServer) GetParameterDetails(c context.Context, mpIDs *pb.ModelParameterIDs) (*pb.ParameterDetails, error) {
   175  	p, _ := peer.FromContext(c)
   176  	clientIP := p.Addr.String()
   177  
   178  	log.Debugln("Got a GetParameterDetails from", clientIP)
   179  	rParameterDetails := &pb.ParameterDetails{}
   180  	s.parameterRegistry.muInfo.RLock()
   181  	defer s.parameterRegistry.muInfo.RUnlock()
   182  
   183  	if len(mpIDs.Ids) == 0 {
   184  		for _, modelDetails := range s.parameterRegistry.ParameterDetail {
   185  			for _, modelDetail := range modelDetails {
   186  				rParameterDetails.Details = append(rParameterDetails.Details, modelDetail)
   187  			}
   188  		}
   189  	} else if len(mpIDs.Ids) == 1 && int(mpIDs.Ids[0].Parameter) == 0 {
   190  		// Return all parameters for model
   191  		if _, exists := s.parameterRegistry.ParameterDetail[mpIDs.Ids[0].Model]; exists {
   192  			for _, modelDetail := range s.parameterRegistry.ParameterDetail[mpIDs.Ids[0].Model] {
   193  				rParameterDetails.Details = append(rParameterDetails.Details, modelDetail)
   194  			}
   195  		} else {
   196  			log.Error("Invalid model ID specified")
   197  		}
   198  	} else {
   199  		for _, mpID := range mpIDs.Ids {
   200  			d, err := s.getParameterDetail(mpID)
   201  			if err != nil {
   202  				log.Errorf(err.Error())
   203  				return nil, err
   204  			}
   205  			rParameterDetails.Details = append(rParameterDetails.Details, d)
   206  
   207  		}
   208  	}
   209  	log.Debugf("Send ParameterDetails for %v parameters", len(rParameterDetails.Details))
   210  	return rParameterDetails, nil
   211  }

   213  func (s *IbeamServer) getParameterDetail(mpID *pb.ModelParameterID) (*pb.ParameterDetail, error) {
   214  	if mpID.Parameter == 0 {
   215  		return nil, errors.New("Failed to get instance values " + mpID.String())
   216  	}
   217  	if len(s.parameterRegistry.ParameterDetail) < int(mpID.Model) {
   218  		return nil, fmt.Errorf("ParamerDetail does not have Model with id %v", mpID.Model)
   219  	}
   220  	for _, parameterDetail := range s.parameterRegistry.ParameterDetail[mpID.Model] {
   221  		if parameterDetail.Id.Model == mpID.Model && parameterDetail.Id.Parameter == mpID.Parameter {
   222  			return parameterDetail, nil
   223  		}
   224  	}
   225  	return nil, errors.New("Cannot find ParameterDetail with given ModelParameterID")
   226  }

   236  // Subscribe starts a ServerStream and send updates if a Parameter changes.
   237  // On subscribe all current values should be sent back.
   238  // If no IDs are given, subscribe to every Parameter.
   239  func (s *IbeamServer) Subscribe(dpIDs *pb.DeviceParameterIDs, stream pb.IbeamCore_SubscribeServer) error {
   240  	p, _ := peer.FromContext(stream.Context())
   241  	clientIP := p.Addr.String()
   242  	log.Debug("New Client subscribed from ", clientIP)
   243  
   244  	// Fist send all parameters
   245  	parameters, err := s.Get(nil, dpIDs)
   246  	if err != nil {
   247  		return err
   248  	}
   249  
   250  	for _, parameter := range parameters.Parameters {
   251  		log.Debugf("Send Parameter with ID '%v' to client", parameter.Id)
   252  		log.Debugf("Param: %+v", parameter)
   253  		stream.Send(parameter)
   254  	}
   255  
   256  	distributor := make(chan *pb.Parameter, 100)
   257  	s.muDistributor.Lock()
   258  	s.serverClientsDistributor[distributor] = true
   259  
   260  	log.Debugf("Added distributor number %v", len(s.serverClientsDistributor))
   261  	s.muDistributor.Unlock()
   262  
   263  	ping := time.NewTicker(time.Second / 2)
   264  	for {
   265  		select {
   266  		case <-ping.C:
   267  			err := stream.Context().Err()
   268  			if err != nil {
   269  				s.muDistributor.Lock()
   270  				delete(s.serverClientsDistributor, distributor)
   271  				s.muDistributor.Unlock()
   272  				log.Warn("Connection to client for subscription lost")
   273  				return nil
   274  			}
   275  		case parameter := <-distributor:
   276  			if parameter.Id == nil || parameter.Id.Device == 0 || parameter.Id.Parameter == 0 {
   277  				continue
   278  			}
   279  			// Check if Device is Subscribed
   280  			if len(dpIDs.Ids) == 1 && dpIDs.Ids[0].Parameter == 0 && dpIDs.Ids[0].Device != parameter.Id.Device {
   281  				log.Tracef("Blocked sending out of change because of devicefilter Want: %d, Got: %d", dpIDs.Ids[0].Device, parameter.Id.Device)
   282  				continue
   283  			}
   284  
   285  			// Check for parameter filtering
   286  			if len(dpIDs.Ids) >= 1 && dpIDs.Ids[0].Parameter != 0 && !containsDeviceParameter(parameter.Id, dpIDs) {
   287  				log.Tracef("Blocked sending out of change of parameter %d (D: %d) because of device parameter id filter, %v", parameter.Id.Parameter, parameter.Id.Device, dpIDs)
   288  				continue
   289  			}
   290  
   291  			log.Debugf("Send Parameter with ID '%v' to client from ServerClientsStream", parameter.Id)
   292  			stream.Send(parameter)
   293  		}
   294  	}
   295  }

   302  	}
   303  	return false

   315  // CreateServerWithDefaultModel sets up the ibeam server, parameter manager and parameter registry and allows to specify a default model
   316  func CreateServerWithDefaultModel(coreInfo *pb.CoreInfo, defaultModel *pb.ModelInfo) (manager *IbeamParameterManager, registry *IbeamParameterRegistry, settoManager chan *pb.Parameter, getfromManager chan *pb.Parameter) {
   317  	clientsSetter := make(chan *pb.Parameter, 100)
   318  	getfromManager = make(chan *pb.Parameter, 100)
   319  	settoManager = make(chan *pb.Parameter, 100)
   320  
   321  	fistParameter := pb.Parameter{}
   322  	fistParameter.Id = &pb.DeviceParameterID{
   323  		Device:    0,
   324  		Parameter: 0,
   325  	}
   326  
   327  	watcher := make(chan *pb.Parameter)
   328  
   329  	coreInfo.IbeamVersion = pb.File_ibeam_core_proto.Options().ProtoReflect().Get(pb.E_IbeamVersion.TypeDescriptor()).String()
   330  
   331  	registry = &IbeamParameterRegistry{
   332  		coreInfo:        proto.Clone(coreInfo).(*pb.CoreInfo),
   333  		DeviceInfos:     map[uint32]*pb.DeviceInfo{},
   334  		ModelInfos:      map[uint32]*pb.ModelInfo{},
   335  		ParameterDetail: map[uint32]map[uint32]*pb.ParameterDetail{},
   336  		ParameterValue:  map[uint32]map[uint32]*IbeamParameterDimension{},
   337  	}
   338  
   339  	server := IbeamServer{
   340  		parameterRegistry:        registry,
   341  		clientsSetterStream:      clientsSetter,
   342  		serverClientsStream:      watcher,
   343  		serverClientsDistributor: make(map[chan *pb.Parameter]bool),
   344  	}
   345  
   346  	manager = &IbeamParameterManager{
   347  		parameterRegistry:   registry,
   348  		out:                 getfromManager,
   349  		in:                  settoManager,
   350  		clientsSetterStream: clientsSetter,
   351  		serverClientsStream: watcher,
   352  		server:              &server,
   353  	}
   354  
   355  	go func() {
   356  		for {
   357  			parameter := <-watcher
   358  			server.muDistributor.RLock()
   359  			for channel, isOpen := range server.serverClientsDistributor {
   360  				if isOpen {
   361  					channel <- parameter
   362  				} else {
   363  					log.Debugf("Deleted Channel %v", channel)
   364  					server.muDistributor.RUnlock()
   365  					server.muDistributor.Lock()
   366  					delete(server.serverClientsDistributor, channel)
   367  					server.muDistributor.Unlock()
   368  					server.muDistributor.RLock()
   369  				}
   370  			}
   371  			server.muDistributor.RUnlock()
   372  		}
   373  	}()
   374  	registry.RegisterModel(defaultModel)
   375  	return
   376  }
